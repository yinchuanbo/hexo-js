---
title: "webpack5-修改输出文件目录"
date: 2024-01-07T17:49:10+08:00
---

```js
const path = require("path");
module.exports = {
  entry: "./src/main.js",
  output: {
    // 所有文件的打包输出目录
    path: path.resolve(__dirname, "dist"),
    // 入口文件打包输出的文件名
    filename: "static/js/main.js",
  },
  // 加载器
  module: {
    rules: [
      //...
      {
        test: /\.(png|jpe?g|gif|webp)$/,
        type: "asset",
        parser: {
          dataUrlCondition: {
            maxSize: 10 * 1024,
          },
        },
        generator: {
          // 修改图片的输出路径和文件名
          filename: "static/images/[hash:10][ext][query]",
        },
      },
    ],
  },
  // 插件
  plugins: [],
  // 模式
  mode: "development",
};
```
