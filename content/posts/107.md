```js
const path = require("path");
module.exports = {
  entry: "./src/main.js",
  output: {
    // 所有文件的打包输出目录
    path: path.resolve(__dirname, "dist"),
    // 入口文件打包输出的文件名
    filename: "static/js/main.js",
    clean: true; // 自动清空上次打包结果
  },
  // 加载器
  module: {
    rules: [
      //...
      {
        test: /\.(ttf|woff2)$/,
        type: "asset/resource",
        generator: {
          filename: "static/media/[hash:10][ext][query]",
        },
      },
    ],
  },
  // 插件
  plugins: [],
  // 模式
  mode: "development",
};
```
